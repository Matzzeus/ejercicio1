Algoritmo sin_titulo
	Escribir 'Ingrese un número entero'
	Leer n1
	pares <- 0
	impares <- 0
	Para i<-1 Hasta n1 Hacer
		random <- azar(100)
		Escribir '',random
		Mientras (random>1) Hacer
			random <- random-2
		FinMientras
		Si (random=0) Entonces
			Escribir 'Es par '
			pares <- pares+1
		SiNo
			Escribir 'Es impar '
			impares <- impares+1
		FinSi
	FinPara
	Escribir 'Son ',pares,' números pares y ',impares,' número impares'
	
FinAlgoritmo
